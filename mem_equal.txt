#include <stdio.h>

#define MAX_PROCESSES 100

typedef struct {
    int id;
    int memory;
} Process;

void equalAllocation(int totalMemorySize, int numberOfProcesses, Process processes[]) {
    int memoryPerProcess = totalMemorySize / numberOfProcesses;

    printf("Memory Allocation Table:\n");
    for (int i = 0; i < numberOfProcesses; i++) {
        printf("Process %d: Memory allocated = %d\n", processes[i].id, memoryPerProcess);
    }
}

int main() {
    int totalMemorySize = 1024; // Total memory size in KB
    int numberOfProcesses = 3;
    Process processes[MAX_PROCESSES] = {
        {1, 0}, // Process IDs start from 1
        {2, 0},
        {3, 0}
    };

    for (int i = 0; i < numberOfProcesses; i++) {
        printf("Enter memory requirement for Process %d (in KB): ", i + 1);
        scanf("%d", &processes[i].memory);
    }
    equalAllocation(totalMemorySize, numberOfProcesses, processes);
    return 0;
}
